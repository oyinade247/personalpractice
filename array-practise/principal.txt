def calculate_monthly_payment(principal, annual_interest_rate, years):
    monthly_rate = annual_interest_rate / 100 / 12
    number_of_months = years * 12

    monthly_payment = principal * (
        monthly_rate * (1 + monthly_rate) ** number_of_months
    ) / ((1 + monthly_rate) ** number_of_months - 1)

    return monthly_payment

import java.util.Arrays;
import java.util.Scanner;

public class ParkingLot {
    private int[] slots = new int[20];

    // Display parking status
    public void display() {
        System.out.println(Arrays.toString(slots));
    }

    // Car enters the parking lot
    public void enter() {
        for (int i = 0; i < slots.length; i++) {
            if (slots[i] == 0) {
                slots[i] = 1;
                System.out.println("Car parked at slot " + (i + 1));
                return;
            }
        }
        System.out.println("Parking lot full!");
    }

    // Car leaves the parking lot
    public void leave(int slotNumber) {
        if (slotNumber < 1 || slotNumber > 20) {
            System.out.println("Invalid slot number!");
            return;
        }
        if (slots[slotNumber - 1] == 1) {
            slots[slotNumber - 1] = 0;
            System.out.println("Car left slot " + slotNumber);
        } else {
            System.out.println("Slot " + slotNumber + " is already empty!");
        }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        ParkingLot lot = new ParkingLot();
        String command;

        System.out.println("Parking lot simulation started.");
        lot.display();

        while (true) {
            System.out.println("\nEnter command (enter / leave <slot> / display / exit):");
            command = sc.nextLine().trim().toLowerCase();

            if (command.equals("exit")) {
                break;
            } else if (command.equals("enter")) {
                lot.enter();
            } else if (command.startsWith("leave")) {
                try {
                    int slot = Integer.parseInt(command.split(" ")[1]);
                    lot.leave(slot);
                } catch (Exception e) {
                    System.out.println("Invalid command. Use leave <slot_number>");
                }
            } else if (command.equals("display")) {
                lot.display();
            } else {
                System.out.println("Unknown command!");
            }
        }

        sc.close();
        System.out.println("Simulation ended.");
    }
}

import java.util.Scanner;

public class CreditCardValidation {

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);

        // Ask the user to enter a credit card number
        System.out.print("Enter a credit card number as a long integer: ");
        long number = input.nextLong();

        // Display whether the number is valid
        if (isValid(number))
            System.out.println(number + " is valid.");
        else
            System.out.println(number + " is invalid.");

        input.close();
    }

    /** Return true if the card number is valid */
    public static boolean isValid(long number) {
        return (getSize(number) >= 13 && getSize(number) <= 16) &&
               (prefixMatched(number, 4) ||
                prefixMatched(number, 5) ||
                prefixMatched(number, 37) ||
                prefixMatched(number, 6)) &&
               ((sumOfDoubleEvenPlace(number) + sumOfOddPlace(number)) % 10 == 0);
    }

    /** Get the result from Step 2: Double every second digit from right to left */
    public static int sumOfDoubleEvenPlace(long number) {
        int sum = 0;
        String num = number + "";

        for (int i = getSize(number) - 2; i >= 0; i -= 2) {
            int digit = Integer.parseInt(num.charAt(i) + "");
            sum += getDigit(digit * 2);
        }
        return sum;
    }

    /** Return this number if it is a single digit, otherwise return sum of the two digits */
    public static int getDigit(int number) {
        if (number < 10)
            return number;
        else
            return (number / 10) + (number % 10);
    }

    /** Return sum of digits in the odd places from right to left in the card number */
    public static int sumOfOddPlace(long number) {
        int sum = 0;
        String num = number + "";

        for (int i = getSize(number) - 1; i >= 0; i -= 2) {
            int digit = Integer.parseInt(num.charAt(i) + "");
            sum += digit;
        }
        return sum;
    }

    /** Return true if the digit d is a prefix for number */
    public static boolean prefixMatched(long number, int d) {
        return getPrefix(number, getSize(d)) == d;
    }

    /** Return the number of digits in d */
    public static int getSize(long d) {
        String num = d + "";
        return num.length();
    }

    /** Return the first k number of digits from number */
    public static long getPrefix(long number, int k) {
        String num = number + "";
        if (num.length() > k)
            return Long.parseLong(num.substring(0, k));
        return number;
    }
}

def calculate_invoice(customer):
    """Calculate totals, discount, VAT, and grand total for one customer."""
    products = customer["products"]
    prices = customer["prices"]
    quantities = customer["quantities"]

    totals = []
    subtotal = 0

    # Calculate totals for each product
    for i in range(len(products)):
        total = prices[i] * quantities[i]
        totals.append(total)
        subtotal += total

    discount = subtotal * 0.10 if subtotal >= 50000 else 0
    vat = subtotal * 0.075
    grand_total = subtotal - discount + vat

    return totals, subtotal, discount, vat, grand_total


def display_invoice(customer):
    """Display invoice for one customer."""
    totals, subtotal, discount, vat, grand_total = calculate_invoice(customer)

    print(f"\nInvoice for {customer['name']}")
    print(f"{'Product':<20}{'Qty':<10}{'Price':<10}{'Total':<10}")
    print("-" * 50)

    for i in range(len(customer["products"])):

def park_car(lot):
    """Park a car in the first available slot. Return slot number or -1 if full."""
    for i in range(len(lot)):
        if lot[i] == 0:
            lot[i] = 1
            return i + 1
    return -1  # Parking lot full

def leave_car(lot, slot_number):
    """Remove a car from a specified slot. Return True if successful, False otherwise."""
    if 1 <= slot_number <= len(lot):
        if lot[slot_number - 1] == 1:
            lot[slot_number - 1] = 0
            return True
        else:
            return False  # Slot already empty
    else:
        return False  # Invalid slot

def get_parking_status(lot):
    """Return a list of slot statuses ('Empty' or 'Occupied')."""
    return ["Empty" if slot == 0 else "Occupied" for slot in lot]

# Function to get the number of digits
def get_size(num):
    return len(str(num))  # Simply count digits

# Function to get the first k digits of a number
def get_prefix(num, k):
    num_str = str(num)
    if len(num_str) < k:
        return num
    else:
        return int(num_str[:k])  # Slice first k characters

# Function to check if prefix matches
def prefix_matched(num, d):
    return get_prefix(num, len(str(d))) == d

# Function to handle doubling a digit
def get_digit(number):
    if number < 10:
        return number
    else:
        return number // 10 + number % 10  # Sum of two digits

# Step 2: Sum of double even-place digits from right
def sum_of_double_even_place(digits):
    sum_even = 0
    # Start from second-to-last digit and step backwards by 2
    for i in range(len(digits)-2, -1, -2):
        sum_even += get_digit(digits[i]*2)
    return sum_even

# Step 3: Sum of odd-place digits from right
def sum_of_odd_place(digits):
    sum_odd = 0
    # Start from last digit and step backwards by 2
    for i in range(len(digits)-1, -1, -2):
        sum_odd += digits[i]
    return sum_odd

# Function to check validity using all rules
def is_valid(number):
    size = get_size(number)
    # Convert to digits list for Luhn algorithm
    digits = [int(d) for d in str(number)]
    
    # Check length and prefix rules
    if size >= 13 and size <= 16 and (
        prefix_matched(number, 4) or
        prefix_matched(number, 5) or
        prefix_matched(number, 6) or
        prefix_matched(number, 37)
    ):
        total = sum_of_double_even_place(digits) + sum_of_odd_place(digits)
        return total % 10 == 0  # Valid if divisible by 10
    else:
        return False

public class EmirpShort {
    public static void main(String[] args) {
        for (int num = 10; num <= 1000; num++) {
            if (isPrime(num)) {
                int rev = reverse(num);
                if (rev != num && isPrime(rev)) {
                    System.out.print(num + " ");
                }
            }
        }
    }

    public static boolean isPrime(int n) {
        if (n < 2) return false;
        for (int i = 2; i <= n / 2; i++) {
            if (n % i == 0) return false;
        }
        return true;
    }

    public static int reverse(int n) {
        int rev = 0;
        while (n > 0) {
            rev = rev * 10 + n % 10;
            n /= 10;
        }
        return rev;
    }
}



def is_anagram(word1, word2):
    # Make both lowercase and remove spaces
    word1 = word1.replace(" ", "").lower()
    word2 = word2.replace(" ", "").lower()

    # Quick check — if lengths differ, they can’t be anagrams
    if len(word1) != len(word2):
        return False

    # Convert strings into lists of characters
    list1 = list(word1)
    list2 = list(word2)

    # Sort both lists
    list1.sort()
    list2.sort()

    # Compare the sorted lists
    return list1 == list2

            w1 += ch.lower()  # or manually convert

        if ch != " ":
            w2 += ch.lower()
    
    # if lengths differ
    if len(w1) != len(w2):

    # make a frequency table manually
    freq1 = {}
    freq2 = {}

public class CapitalizeFirst {

    public static void main(String[] args) {
        String[] words = {"hello", "world", "java", "is", "fun"};
        String[] capitalized = capitalizeFirstLetter(words);

        for (String word : capitalized) {
            System.out.println(word);
        }
    }

    public static String[] capitalizeFirstLetter(String[] words) {
        String[] result = new String[words.length];

        for (int i = 0; i < words.length; i++) {
            if (words[i].length() == 0) {
                result[i] = words[i]; // empty string stays empty
            } else {
                // Use substring and toUpperCase
                result[i] = words[i].substring(0, 1).toUpperCase() + words[i].substring(1);
            }
        }

        return result;
    }
}





import java.util.Arrays;

public class SortByLength {

    public static void main(String[] args) {
        String[] words = {"apple", "dog", "banana", "cat", "elephant"};
        String[] sorted = sortByLength(words);

        System.out.println(Arrays.toString(sorted));
    }

    public static String[] sortByLength(String[] words) {
        String[] result = Arrays.copyOf(words, words.length); // copy original array

        // Simple Bubble Sort based on length
        for (int i = 0; i < result.length - 1; i++) {
            for (int j = 0; j < result.length - 1 - i; j++) {
                if (result[j].length() > result[j + 1].length()) {
                    // Swap words
                    String temp = result[j];
                    result[j] = result[j + 1];
                    result[j + 1] = temp;
                }
            }
        }

        return result;
    }
}


def sort_by_length(words):
    # Manual copy using a for loop
    result = []
    for word in words:
        result.append(word)
    
    # Bubble sort based on length
    for i in range(len(result) - 1):
        for j in range(len(result) - 1 - i):
            if len(result[j]) > len(result[j + 1]):
                # Swap
                result[j], result[j + 1] = result[j + 1], result[j]
    
    return result

import datetime

# Dictionary to store movies and their ratings
movies = {}

def add_movie():
    name = input("Enter the movie name: ").strip()
    if name in movies:
        print(f"Movie '{name}' already exists!")
    else:
        movies[name] = {"ratings": [], "added_on": datetime.datetime.now()}
        print(f"Movie '{name}' added on {movies[name]['added_on'].strftime('%Y-%m-%d %H:%M:%S')}!")

def rate_movie():
    name = input("Enter the movie name: ").strip()
    if name not in movies:
        print(f"Movie '{name}' does not exist! Add it first.")
        return
    try:
        rating = float(input("Enter your rating (1-5): "))
        if rating < 1 or rating > 5:
            print("Rating must be between 1 and 5.")
            return
    except ValueError:
        print("Invalid rating. Enter a number between 1 and 5.")
        return
    movies[name]["ratings"].append(rating)
    print(f"Rating {rating} added for '{name}'!")

def view_average_rating():
    if not movies:
        print("No movies available.")
        return
    print("Average Ratings:")
    for name, info in movies.items():
        if info["ratings"]:
            avg = sum(info["ratings"]) / len(info["ratings"])
            print(f"- {name}: {avg:.2f}")
        else:
            print(f"- {name}: No ratings yet")

def main():
    while True:
        print("\n1. Add a Movie")

        print("4. Exit")
        choice = input("Enter your choice: ").strip()

        if choice == "1":
            add_movie()
        elif choice == "2":
            rate_movie()
        elif choice == "3":
            view_average_rating()
        elif choice == "4":
            print("Exiting the application. Goodbye!")
            break
        else:

main()
